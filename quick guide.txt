 For those of you don't want to go through the entire README just for a quick lookup, here is a list
of all the commands and their formats. Arguments are in brackets, arguments with a ... at the end
signify a non-constant number of arguments. The clauses do not include any ... since the argument
number changes due to evaluation. Some commands have more formats than are listed here, and are
marked with a *.

Commands:
	left		[x]	[y]
	right		[x]	[y]
	move		[x]	[y]
*	scroll		[direction]
*	pause		[time]
*	pausenot	[params...]
	jump		[label]
	call		[label]
	jumpback
	run		[script]	[input...]
*	create		[type]	[name]	[value...]
*	set		[name]	[value...]
	setargto	[prefix]	[name]
	+=		[name]	[value]
	-=		[name]	[value]
	*=		[name]	[value]
	/=		[name]	[value]
	%=		[name]	[value]
	if		[clause...]
	for		[name]	[init]	[max]	[increment]
	end
	break
	return
	quit
	type		.[text]
	press		[button]
	release		[button]
*	optionon	[setting]
	optionoff	[setting]
	print		.[text]
*	println
	printval	[name]

Clauses:
	true
	false
	[boolean]
	[number]
	input		[arg]
	>=		[val1]	[val2]
	>		[val1]	[val2]
	<=		[val1]	[val2]
	<		[val1]	[val2]
*	colorat		[x]	[y]	[r]	[g]	[b]
*	scan		[x]	[y]	[w]	[h]	[r]	[g]	[b]
	not		[clause]
	or		[clause1]	[clause2]
	and		[clause1]	[clause2]
	==		[clause1]	[clause2]
	!=		[clause1]	[clause2]
	+		[val1]	[val2]
	-		[val1]	[val2]
	*		[val1]	[val2]
	/		[val1]	[val2]
	%		[val1]	[val2]

Other:
	:[label]

	if	[clause...]
	[
		[commands]
		...
	]
